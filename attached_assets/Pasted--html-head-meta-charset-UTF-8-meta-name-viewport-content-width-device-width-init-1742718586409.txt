    <script>
        function vibrateDevice() {
            if ("vibrate" in navigator) {
                navigator.vibrate(200);
            }
        }

        const notification = document.getElementById('notification');

        function showNotification() {
            notification.style.display = 'block';
            vibrateDevice();
            setTimeout(() => {
                hideNotification();
            }, 15000);  // Changed to 15 seconds
        }

        function hideNotification() {
            notification.classList.add('hide');
            setTimeout(() => {
                notification.style.display = 'none';
            }, 300);
        }

        let startX, startY, moveX, moveY;

        function addSwipeToHide() {
            notification.addEventListener('touchstart', (e) => {
                startX = e.touches[0].clientX;
                startY = e.touches[0].clientY;
            });

            notification.addEventListener('touchmove', (e) => {
                moveX = e.touches[0].clientX;
                moveY = e.touches[0].clientY;
                
                const distanceX = moveX - startX;
                const distanceY = moveY - startY;

                if (Math.abs(distanceX) > 50 || distanceY < -50) {
                    hideNotification();
                }
            });
        }

        addSwipeToHide();

        // Show notification after 6 seconds
        setTimeout(showNotification, 6000);
    </script>
